                    -- kind of showcase :D
                    local function ProjectPoints()
                      local M = Untils.GetCameraIntrinsicProjectionMatrix(90,script.Parent.AbsoluteSize.Y,script.Parent.AbsoluteSize.X)
                      local Points = {}
                      local D3Points = {}
                      local Offsets = {}
                      local RotationLookVector = CamRot
                      local Target = CamPos+RotationLookVector
                      local Cammatrix = Untils.LookAtMatrix(CamPos,Target,Vector3.new(0,1,0))
                      local MatrixView = MatrixLib.inverseMatrix(Cammatrix)
                      for ind,Object:Obj in pairs(Scene) do
                        if math.abs((Object.Position-CamPos).Magnitude) > RenderDistance then
                          continue
                        end
                        local RX,RY,RZ = Untils.GetRotationMatrix(Object.Rotation)
                        Points[ind] = {}
                        D3Points[ind] = {}
                        Offsets[ind] = {}
                        for i,point in pairs(Object.Points) do
                          -- Project points--
                          local RXM = MatrixLib.MNVMultiply(RX,point)
                          local RXYM = MatrixLib.Multiply(RY,RXM)
                          local RXYZM = MatrixLib.Multiply(RZ,RXYM)
                          local Pos = MatrixLib.MatrixToVec(RXYZM)
                          
                          local WorldPos = Object.Position+Pos+CamPos
                          local Matrix = MatrixLib.Vec3ToMatrix4By1(WorldPos)
                          local ret = MatrixLib.Multiply(MatrixView,Matrix)
                          local NP = Vector3.new(ret[1][1],ret[2][1],ret[3][1])
                          D3Points[ind][i]=NP
  
                          local Proj = Untils.MatrixMultiplicationMatrix4by1Project(M,NP)
                          local V2Pos:Vector2 = Vector2.new((Proj[1][1]+1)/2*script.Parent.AbsoluteSize.X,(Proj[2][1]+1)/2*script.Parent.AbsoluteSize.Y)
                          Points[ind][i]= V2Pos
                          Offsets[ind][i] = Proj[3][1]
                        end
                      end
                      return Points,D3Points
                    end
